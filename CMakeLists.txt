cmake_minimum_required(VERSION 3.22)
project(occ_test)

target_compile_definitions(occ_test PRIVATE "-DSRCDIR=\"${SRC_DIR}\"")
target_link_libraries(occ_test PRIVATE "${OpenCASCADE_LIBRARIES}"
        ${OPENGL_LIBRARIES})
